#include <iostream>
using namespace std;
int linear(int arr[],int n,int key)
{
   int low=0 ;
   int high=n;
   while(low<high)
   {
       int mid=(low+high)/2;
       if(arr[mid]==key)
       {
           cout<<"element found at position "<<mid+1;
           break;
       }
     
       else if(arr[mid]>key)
       {
           high=mid-1;
           continue;
       }
       
       else if(arr[mid]<key)
       {
           low=mid+1;
           continue;
       }
       
       else
       {
           cout<<"element not found";
       }
   }
}

int main()
{
    int n;
    int key;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    cout<<"enter the key to find:"<<endl;
    cin>>key;
   
   
    linear(arr,n,key);
   
    return 0;
}
